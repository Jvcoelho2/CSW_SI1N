import pandas as pd
        import numpy as np
        
        # Exemplo de dataframe com valores que são outliers
        df = pd.DataFrame({
            'valor': [10, 12, 13, 15, 16, 100, 20, 22, 23, 24]
        })
        
        # Calcular o Q1, Q3 e o IQR
        Q1 = df['valor'].quantile(0.25)
        Q3 = df['valor'].quantile(0.75)
        IQR = Q3 - Q1
        
        # Identificar outliers
        outliers = df[(df['valor'] < (Q1 - 1.5 * IQR)) | (df['valor'] > (Q3 + 1.5 * IQR))]
        
        # Remover outliers
        df_clean = df[~((df['valor'] < (Q1 - 1.5 * IQR)) | (df['valor'] > (Q3 + 1.5 * IQR)))]
        
        # Substituir outliers pela mediana
        mediana = df['valor'].median()
        df['valor_substituido'] = np.where(
            (df['valor'] < (Q1 - 1.5 * IQR)) | (df['valor'] > (Q3 + 1.5 * IQR)),
            mediana,
            df['valor']
        )
        
        # Exibir resultados
        print("Outliers identificados:\n", outliers)
        print("\nDataframe sem outliers:\n", df_clean)
        print("\nDataframe com outliers substituídos pela mediana:\n", df)
